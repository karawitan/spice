49 DBusError *error)
50{
51 int written;
52
53 written = _dbus_write (pipe->fd, buffer, start, len);
54 if (written < 0)
55 {
56 dbus_set_error (error, DBUS_ERROR_FAILED,
57 "Writing to pipe: %s\n",
58 _dbus_strerror (errno));
59 }
60 return written;
61}
